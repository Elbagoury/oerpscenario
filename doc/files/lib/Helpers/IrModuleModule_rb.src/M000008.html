      def self.install_modules(modules)
        update=false
        res=true
        modules.each do |m|
          if m.state != 'installed':
            m.state='to install'
            m.save
            update=true
          end
        end
        if update:
          res = IrModuleModule.update_needed_modules()
        end
        if res :
          return true
        else
          raise "!!! --- HELPER ERROR : install_modules was unable to upgrade needed modules.."
        end
      end
