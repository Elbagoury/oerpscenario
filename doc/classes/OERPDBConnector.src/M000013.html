    def setConnexion(user, password, host, database, dbport, port=5432 )
        @port = port
        @user = user
        @dbname = database
        @pwd = password 
        @host = host
        ActiveRecord::Base.establish_connection(:adapter => 'postgresql',
        :host => @host,
        :username => @user,
        :database =>  @dbname,
        :password => @pwd,
        :port =>  @port)
    end
