    def setConnexion(user, password, host, port, database )
       
        @port = port
        @user = user
        @dbname = database
        @pwd = password 
        @host = host
        @common_server = XMLRPC::Client.new(@host, "/xmlrpc/common", @port)
        @object_server = XMLRPC::Client.new(@host, "/xmlrpc/object", @port)
    end
